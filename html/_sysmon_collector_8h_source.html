101 static void WINAPI OnEventRecord(PEVENT_RECORD pEvent) {
102 DWORD size = 0;
103 if (TdhGetEventInformation(pEvent, 0, NULL, NULL, &size) == ERROR_INSUFFICIENT_BUFFER) {
104 std::vector<BYTE> info(size);

117 static void ParseAndLog(PEVENT_RECORD pEvent, PTRACE_EVENT_INFO pInfo) {
118 switch (pEvent->EventHeader.EventDescriptor.Id) {
119 case 3: { // Network connection
120 std::wstring path = GetEventProperty(pEvent, pInfo, L"Image");
121 DWORD port = GetEventPropertyInt(pEvent, L"DestinationPort");
122 std::wcout << L"[NET] " << path << L" Port: " << port << std::endl;
123 break;
124 }
125 case 1: { // Process creation
126 std::wstring cmd = GetEventProperty(pEvent, pInfo, L"CommandLine");
127 double ent = EventFeature::CalculateEntropy(cmd);
128 std::wcout << L"[PROC] Entropy: " << ent << L" CMD: " << cmd << std::endl;
129 break;
130 }
131 }
132 }
