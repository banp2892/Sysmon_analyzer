24 static double CalculateEntropy(const std::wstring& s) {
25 if (s.empty()) return 0.0;
26
27 std::map<wchar_t, int> counts;
28 for (size_t i = 0; i < s.size(); ++i) {
29 counts[s[i]]++;
30 }
31
32 double ent = 0.0;
33 // Используем итератор вместо [ch, count], чтобы не требовать C++17
34 for (std::map<wchar_t, int>::const_iterator it = counts.begin(); it != counts.end(); ++it) {
35 double p = (double)it->second / s.size();
36 ent -= p * (log(p) / log(2.0)); // log2 может не быть в старых VS, это замена
37 }
38 return ent;
39 }
